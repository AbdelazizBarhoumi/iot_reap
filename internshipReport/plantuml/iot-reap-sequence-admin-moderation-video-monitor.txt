== Video Monitoring ==

admin -> frontend : Navigate to video monitoring
frontend -> routes : GET /admin/video-monitor
routes -> frontend : Render video monitor page
frontend --> admin : Show video monitoring dashboard

admin -> frontend : View active camera streams
frontend -> routes : GET /admin/cameras/active
routes -> frontend : Load active cameras
frontend -> db : Query active camera sessions
db --> frontend : Return active cameras
frontend --> admin : Display live camera streams

admin -> frontend : Monitor specific camera
frontend -> routes : GET /admin/cameras/{camera}/monitor
routes -> frontend : Load camera stream
frontend -> db : Query camera details
frontend -> db : Get stream URL
db --> frontend : Return camera data
frontend --> admin : Display camera stream with controls

admin -> frontend : Take camera control
frontend -> routes : POST /admin/cameras/{camera}/control
routes -> frontend : Acquire admin control
frontend -> db : Update camera control status
db --> frontend : Confirm control
frontend --> admin : Enable PTZ controls

admin -> frontend : Send PTZ command
frontend -> routes : POST /admin/cameras/{camera}/move
routes -> frontend : Send PTZ movement
frontend -> db : Update camera position
db --> frontend : Confirm movement
frontend --> admin : Show camera movement

admin -> frontend : Release camera control
frontend -> routes : DELETE /admin/cameras/{camera}/control
routes -> frontend : Release admin control
frontend -> db : Update camera control status
db --> frontend : Confirm release
frontend --> admin : Disable PTZ controls

admin -> frontend : View camera recording
frontend -> routes : GET /admin/cameras/{camera}/recording
routes -> frontend : Load camera recording
frontend -> db : Query recording files
db --> frontend : Return recording list
frontend --> admin : Display available recordings

admin -> frontend : Download recording
frontend -> routes : GET /admin/cameras/{camera}/recording/{recordingId}
routes -> frontend : Stream recording file
frontend -> db : Load recording file
db --> frontend : Return file stream
frontend --> admin : Download recording file

admin -> frontend : View camera analytics
frontend -> routes : GET /admin/cameras/{camera}/analytics
routes -> frontend : Load camera analytics
frontend -> db : Query camera usage statistics
frontend -> db : Calculate viewing metrics
db --> frontend : Return analytics data
frontend --> admin : Display camera usage analytics

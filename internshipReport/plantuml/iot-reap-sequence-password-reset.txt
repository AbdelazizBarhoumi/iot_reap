@startuml iot-reap-sequence-password-reset
title IoT-REAP - Password Reset Sequence Diagram
actor Guest as guest
participant "Frontend\n(React/Inertia)" as frontend
participant "Laravel\nRoutes" as routes
participant "Auth\nController" as auth
participant "User\nModel" as user
participant "Database" as db
participant "Email\nService" as email

guest -> frontend : Request password reset
frontend -> routes : POST /auth/forgot-password
routes -> auth : forgotPassword(email)
auth -> user : findByEmail(email)
user --> auth : Return user or null

alt User found
    auth -> user : generatePasswordResetToken()
    user -> db : Save reset token and expiry
    auth -> email : sendPasswordResetEmail(user)
    email --> guest : Send reset email with link
    auth --> routes : Return success response
    routes --> frontend : Display success message
    frontend --> guest : Show "Check your email" message

    guest -> frontend : Click reset link from email
    frontend -> routes : GET /reset-password/{token}
    routes --> frontend : Render reset password form
    frontend --> guest : Display password reset form

    guest -> frontend : Submit new password
    frontend -> routes : POST /auth/reset-password
    routes -> auth : resetPassword(token, password)
    auth -> user : findByResetToken(token)
    user --> auth : Return user or null

    alt Valid token
        auth -> user : updatePassword(password)
        user -> db : Save new password
        auth -> user : clearResetToken()
        user -> db : Clear reset token
        auth --> routes : Return success
        routes --> frontend : Redirect to login
        frontend --> guest : Display login page with success
    else Invalid/expired token
        auth --> routes : Return error
        routes --> frontend : Display error message
        frontend --> guest : Show "Invalid or expired link" error
    end
else User not found
    auth --> routes : Return success (security)
    routes --> frontend : Display success message
    frontend --> guest : Show "Check your email" (even if user doesn't exist)
end

note right of guest
  Password reset flow:
  - Rate limited (5 attempts per minute)
  - Token expires after configurable time
  - Security: Always return success to prevent email enumeration
  - User must click email link to reset
end note

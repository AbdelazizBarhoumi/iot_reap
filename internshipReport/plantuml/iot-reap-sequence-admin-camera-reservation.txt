actor Admin as admin
participant "Frontend\n(React/Inertia)" as frontend
participant "Laravel\nRoutes" as routes
participant "Admin Camera\nController" as camera
participant "Admin Reservation\nController" as reservation
participant "Database" as db
participant "Camera\nModel" as cm
participant "Reservation\nModel" as rm
participant "Gateway\nNodes" as gateways

== Camera Administration ==

admin -> frontend : Navigate to camera management
frontend -> routes : GET /admin/cameras
routes -> camera : cameras()
camera -> db : Load all cameras
camera -> db : Load camera assignments
camera -> db : Load camera status
db --> camera : Return camera data
camera --> routes : Return cameras
routes --> frontend : Display camera list
frontend --> admin : Show all cameras with assignments

admin -> frontend : Assign camera to VM
frontend -> routes : PUT /admin/cameras/{camera}/assign
routes -> camera : assignToVm(cameraId, vmId)
camera -> db : Load camera record
camera -> db : Load VM record
camera -> db : Create camera-VM assignment
camera -> db : Save changes
camera --> routes : Return assignment
routes --> frontend : Update camera display
frontend --> admin : Show camera assigned to VM

admin -> frontend : Unassign camera
frontend -> routes : DELETE /admin/cameras/{camera}/assign
routes -> camera : unassignFromVm(cameraId)
camera -> db : Load camera assignment
camera -> db : Delete assignment record
camera -> db : Save changes
camera --> routes : Return success
routes --> frontend : Update camera display
frontend --> admin : Show camera unassigned

admin -> frontend : Bulk assign cameras
frontend -> routes : POST /admin/cameras/bulk-assign
routes -> camera : bulkAssign(cameraIds, vmId)
camera -> db : Load camera records
camera -> db : Load VM record
loop For each camera
    camera -> db : Create camera-VM assignment
end
camera -> db : Save all changes
camera --> routes : Return assignment count
routes --> frontend : Update camera display
frontend --> admin : Show cameras assigned

admin -> frontend : Activate camera
frontend -> routes : PUT /admin/cameras/{camera}/activate
routes -> camera : activate(cameraId)
camera -> db : Load camera record
camera -> db : Update status to active
camera -> db : Save changes
camera -> gateways : Activate camera stream
gateways --> camera : Confirm activation
camera --> routes : Return success
routes --> frontend : Update camera status
frontend --> admin : Show camera as active

admin -> frontend : Deactivate camera
frontend -> routes : PUT /admin/cameras/{camera}/deactivate
routes -> camera : deactivate(cameraId)
camera -> db : Load camera record
camera -> db : Update status to inactive
camera -> db : Save changes
camera -> gateways : Deactivate camera stream
gateways --> camera : Confirm deactivation
camera --> routes : Return success
routes --> frontend : Update camera status
frontend --> admin : Show camera as inactive

db --> reservation : Return pending reservations
reservation --> routes : Return pending list

routes --> frontend : Display upcoming reservations

admin -> frontend : Approve USB reservation
frontend -> routes : POST /admin/reservations/{reservation}/approve
routes -> reservation : approve(reservationId)
reservation -> db : Load reservation record
reservation -> db : Update status to approved
reservation -> db : Set approved_at timestamp
reservation -> db : Set approved_by to admin

reservation -> db : Notify user

routes --> frontend : Update reservation status

reservation -> db : Load reservation record
reservation -> db : Update status to rejected
reservation -> db : Set rejection reason
reservation -> db : Save changes
reservation -> db : Notify user
reservation --> routes : Return success
routes --> frontend : Update reservation status
frontend --> admin : Show reservation as rejected

admin -> frontend : Block device time slot
frontend -> routes : POST /admin/reservations/block
routes -> reservation : createBlock(blockData)
reservation -> db : Create block reservation
reservation -> db : Set status to blocked
reservation -> db : Save changes
reservation --> routes : Return created block
routes --> frontend : Add block to calendar
frontend --> admin : Show time slot blocked

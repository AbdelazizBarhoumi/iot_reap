entity "users" as users {
  * id : BIGINT <<PK>>
  --
  name : VARCHAR
  email : VARCHAR <<UNIQUE>>
  email_verified_at : TIMESTAMP
  password : VARCHAR
  google_id : VARCHAR <<UNIQUE>>
  avatar : VARCHAR
  role : ENUM(<<engineer, teacher, admin>>)
  is_suspended : BOOLEAN
  created_at : TIMESTAMP
}

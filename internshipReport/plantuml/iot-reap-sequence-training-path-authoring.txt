teacher -> frontend : Navigate to teaching dashboard
frontend -> routes : GET /teaching
routes -> teaching : index()
teaching -> db : Load teacher's training paths

routes --> frontend : Display teaching analytics

routes -> teaching : index()
teaching -> db : Load teacher's training paths

teaching --> routes : Return training paths
routes --> frontend : Display training paths list
frontend --> teacher : Show all training paths with status

routes --> frontend : Render create form
frontend --> teacher : Display training path creation form

teacher -> frontend : Fill form and submit
frontend -> routes : POST /teaching
routes -> teaching : store(trainingPathData)
teaching -> db : Create training path record
teaching -> db : Set status to draft
teaching -> tp : Initialize empty modules array
tp --> teaching : Return new training path
teaching --> routes : Return created path
routes --> frontend : Redirect to edit page
frontend --> teacher : Show training path editor

teacher -> frontend : Edit training path details

routes -> teaching : edit(trainingPathId)

teaching -> db : Load modules and units
db --> teaching : Return full path data
teaching --> routes : Return training path
routes --> frontend : Display edit form

teacher -> frontend : Update training path details
frontend -> routes : PATCH /teaching/{trainingPath}

teaching -> db : Save changes

routes --> frontend : Update display
frontend --> teacher : Show updated training path

teacher -> frontend : Edit module details

routes -> teaching : updateModule(trainingPathId, moduleId, data)

teaching -> db : Save changes

routes --> frontend : Update module display
frontend --> teacher : Show updated module

frontend -> routes : DELETE /teaching/{trainingPath}/modules/{module}

teaching -> db : Save changes
teaching --> routes : Return success
routes --> frontend : Remove module from UI
frontend --> teacher : Show module deleted

teacher -> frontend : Reorder modules
frontend -> routes : PATCH /teaching/{trainingPath}/modules/reorder
routes -> teaching : reorderModules(trainingPathId, moduleOrder)

routes --> frontend : Update module order

frontend -> routes : PATCH /teaching/{trainingPath}/modules/{module}/trainingUnits/{unit}
routes -> teaching : updateTrainingUnit(trainingPathId, moduleId, unitId, data)

teaching -> db : Save changes

frontend -> routes : POST /teaching/{trainingPath}/restore
